\[
    \min\quad48(\sum_{i=1}^{4}x_i) + 84(\sum_{j=1}^4y_j)

\[
    x_1 + y_1 + x_4 + y_4 + y_3 \ge 12

    x_2 + y_2 + x_1 + y_1 + y_4 \ge 8

    x_3 + y_3 + x_2 + y_2 + y_1 \ge 6

    x_4 + y_4 + x_3 + y_3 + y_2 \ge 15

    x_i,y_i \ge 0 \quad\forall i \in [1\dots4]
